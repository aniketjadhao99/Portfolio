import javax.swing.*;
import java.awt.*;

public class TrafficSignal extends JFrame implements Runnable {
    private String color = "RED"; // Initial state
    private Thread t;

    public TrafficSignal() {
        setTitle("Traffic Signal Simulation");
        setSize(200, 500);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLocationRelativeTo(null);
        
        // Start the thread
        t = new Thread(this);
        t.start();
        
        setVisible(true);
    }

    @Override
    public void run() {
        try {
            while (true) {
                // RED State
                color = "RED";
                repaint();
                Thread.sleep(5000); // Wait 5 seconds

                // GREEN State
                color = "GREEN";
                repaint();
                Thread.sleep(5000); // Wait 5 seconds

                // YELLOW State
                color = "YELLOW";
                repaint();
                Thread.sleep(2000); // Wait 2 seconds
            }
        } catch (InterruptedException e) {
            System.out.println(e);
        }
    }

    @Override
    public void paint(Graphics g) {
        super.paint(g);
        
        // Draw the signal box
        g.setColor(Color.BLACK);
        g.fillRect(50, 50, 100, 300);

        // Draw Red Light
        g.setColor(color.equals("RED") ? Color.RED : Color.DARK_GRAY);
        g.fillOval(70, 70, 60, 60);

        // Draw Yellow Light
        g.setColor(color.equals("YELLOW") ? Color.YELLOW : Color.DARK_GRAY);
        g.fillOval(70, 170, 60, 60);

        // Draw Green Light
        g.setColor(color.equals("GREEN") ? Color.GREEN : Color.DARK_GRAY);
        g.fillOval(70, 270, 60, 60);
    }

    public static void main(String[] args) {
        new TrafficSignal();
    }
}

import javax.swing.*;

import java.awt.event.*;

public class VerticalBallMovement extends JFrame implements ActionListener {
    JButton b1;
    BallPanel panel;

    public VerticalBallMovement() {
        setTitle("Vertical Ball Movement");
        setSize(400, 500);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new BorderLayout());

        panel = new BallPanel();
        add(panel, BorderLayout.CENTER);

        b1 = new JButton("Start Ball");
        b1.addActionListener(this);
        add(b1, BorderLayout.SOUTH);

        setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {
        if (e.getSource() == b1) {
            // Create and start a new thread for the ball
            Thread t = new Thread(panel);
            t.start();
            b1.setEnabled(false); // Disable button after starting
        }
    }

    public static void main(String[] args) {
        new VerticalBallMovement();
    }
}

class BallPanel extends JPanel implements Runnable {
    int y = 0; // Vertical position
    int direction = 1; // 1 for down, -1 for up

    @Override
    public void run() {
        try {
            while (true) {
                y += (5 * direction);

                // Reverse direction if ball hits top or bottom boundaries
                if (y <= 0 || y >= getHeight() - 30) {
                    direction *= -1;
                }

                repaint(); // Refresh the panel
                Thread.sleep(30); // Control speed of animation
            }
        } catch (InterruptedException e) {
            System.out.println(e);
        }
    }

    @Override
    protected void paintComponent(Graphics g) {
        super.paintComponent(g);
        g.setColor(Color.RED);
        // Draw the ball at the center horizontally (width/2) and variable y
        g.fillOval(getWidth() / 2 - 15, y, 30, 30);
    }
}

import java.io.*;
import java.util.*;

class FileSearcher extends Thread {
    private String fileName;
    private String searchString;

    public FileSearcher(String fileName, String searchString) {
        this.fileName = fileName;
        this.searchString = searchString;
    }

    @Override
    public void run() {
        try (BufferedReader br = new BufferedReader(new FileReader(fileName))) {
            String line;
            int lineNumber = 1;
            boolean found = false;

            while ((line = br.readLine()) != null) {
                if (line.contains(searchString)) {
                    System.out.println("[FOUND] File: " + fileName + " | Line: " + lineNumber + " | Content: " + line.trim());
                    found = true;
                }
                lineNumber++;
            }
        } catch (IOException e) {
            System.err.println("Error reading file " + fileName + ": " + e.getMessage());
        }
    }
}

public class MultithreadedSearch {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter the string to search: ");
        String searchString = sc.nextLine();

        // Get the current directory
        File currentDir = new File(".");
        
        // Filter for text files
        File[] files = currentDir.listFiles((dir, name) -> name.toLowerCase().endsWith(".txt"));

        if (files == null || files.length == 0) {
            System.out.println("No text files found in the current directory.");
            return;
        }

        System.out.println("Searching in " + files.length + " files...\n");

        // Create and start a thread for each file
        for (File file : files) {
            FileSearcher thread = new FileSearcher(file.getName(), searchString);
            thread.start();
        }
        
        sc.close();
    }
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class CounterThread extends JFrame implements ActionListener {
    private JTextField t1;
    private JButton b1;
    private Thread countThread;
    private boolean running = false;

    public CounterThread() {
        setTitle("Multithreaded Counter");
        setSize(300, 150);
        setLayout(new FlowLayout());
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        t1 = new JTextField(15);
        t1.setEditable(false);
        t1.setFont(new Font("Arial", Font.BOLD, 20));
        
        b1 = new JButton("Start Counter");
        b1.addActionListener(this);

        add(t1);
        add(b1);

        setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {
        // Prevent starting multiple threads if one is already running
        if (!running) {
            running = true;
            b1.setEnabled(false); // Disable button after start
            
            // Create the worker object
            CounterWorker worker = new CounterWorker();
            // Pass it to the Thread constructor (Runnable Interface)
            countThread = new Thread(worker);
            countThread.start();
        }
    }

    // Inner class implementing Runnable
    class CounterWorker implements Runnable {
        @Override
        public void run() {
            try {
                for (int i = 1; i <= 100; i++) {
                    final String val = String.valueOf(i);
                    
                    // Update the UI safely
                    SwingUtilities.invokeLater(new Runnable() {
                        public void run() {
                            t1.setText(val);
                        }
                    });

                    Thread.sleep(100); // Delay of 100ms
                }
                
                // Reset button after finishing
                SwingUtilities.invokeLater(() -> {
                    b1.setEnabled(true);
                    running = false;
                });
                
            } catch (InterruptedException ex) {
                System.out.println(ex);
            }
        }
    }

    public static void main(String[] args) {
        new CounterThread();
    }
}

class TemplePanel extends JPanel {
    private boolean drawBase = false;
    private boolean drawPillars = false;
    private boolean drawTop = false;

    public void startConstruction() {
        // Thread 1: Construct the Base
        new Thread(() -> {
            try {
                Thread.sleep(1000);
                drawBase = true;
                repaint();
            } catch (InterruptedException e) { e.printStackTrace(); }
        }).start();

        // Thread 2: Construct the Pillars
        new Thread(() -> {
            try {
                Thread.sleep(2500); // Waits for base
                drawPillars = true;
                repaint();
            } catch (InterruptedException e) { e.printStackTrace(); }
        }).start();

        // Thread 3: Construct the Shikhara (Top)
        new Thread(() -> {
            try {
                Thread.sleep(4000); // Waits for pillars
                drawTop = true;
                repaint();
            } catch (InterruptedException e) { e.printStackTrace(); }
        }).start();
    }

    @Override
    protected void paintComponent(Graphics g) {
        super.paintComponent(g);
        Graphics2D g2 = (Graphics2D) g;
        g2.setStroke(new BasicStroke(3));

        // Draw Base
        if (drawBase) {
            g2.setColor(new Color(139, 69, 19)); // Brown
            g2.fillRect(100, 350, 300, 50);
            g2.setColor(Color.BLACK);
            g2.drawRect(100, 350, 300, 50);
        }

        // Draw Pillars
        if (drawPillars) {
            g2.setColor(new Color(210, 180, 140)); // Tan
            g2.fillRect(130, 200, 30, 150); // Left pillar
            g2.fillRect(340, 200, 30, 150); // Right pillar
            g2.setColor(Color.BLACK);
            g2.drawRect(130, 200, 30, 150);
            g2.drawRect(340, 200, 30, 150);
        }

        // Draw Top (Triangle/Dome)
        if (drawTop) {
            g2.setColor(Color.ORANGE);
            int[] xPoints = {100, 250, 400};
            int[] yPoints = {200, 50, 200};
            g2.fillPolygon(xPoints, yPoints, 3);
            g2.setColor(Color.BLACK);
            g2.drawPolygon(xPoints, yPoints, 3);
            
            // Draw a small flag
            g2.setColor(Color.RED);
            g2.fillRect(250, 20, 30, 20);
            g2.drawLine(250, 50, 250, 20);
        }
    }
}

import java.util.Scanner;

// The Shared Resource class
class Q {
    int n;
    boolean valueSet = false;

    // Method for Consumer
    synchronized int get() {
        while (!valueSet) {
            try {
                wait(); // Wait if producer hasn't produced
            } catch (InterruptedException e) {
                System.out.println(e);
            }
        }
        System.out.println("Consumed: " + n);
        valueSet = false;
        notify(); // Tell producer to produce next
        return n;
    }

    // Method for Producer
    synchronized void put(int n) {
        while (valueSet) {
            try {
                wait(); // Wait if consumer hasn't consumed yet
            } catch (InterruptedException e) {
                System.out.println(e);
            }
        }
        this.n = n;
        valueSet = true;
        System.out.println("Produced: " + n);
        notify(); // Tell consumer value is ready
    }
}

class Producer implements Runnable {
    Q q;

    Producer(Q q) {
        this.q = q;
        new Thread(this, "Producer").start();
    }

    public void run() {
        int i = 1;
        while (i <= 10) { // Produce 10 values
            q.put(i++);
            try { Thread.sleep(500); } catch (Exception e) {}
        }
    }
}

class Consumer implements Runnable {
    Q q;

    Consumer(Q q) {
        this.q = q;
        new Thread(this, "Consumer").start();
    }

    public void run() {
        while (true) {
            int val = q.get();
            if (val == 10) break; // Stop after 10th value
            try { Thread.sleep(1000); } catch (Exception e) {}
        }
    }
}

public class ProducerConsumerTest {
    public static void main(String[] args) {
        Q q = new Q();
        new Producer(q);
        new Consumer(q);
    }
}

import javax.swing.*;
import java.awt.*;

public class ScrollingText extends JFrame implements Runnable {
    private String message = "Java Multithreading - Scrolling Text";
    private int x = 0; // Starting x-position
    private int y = 100; // Constant y-position
    private Thread t;

    public ScrollingText() {
        setTitle("Scrolling Text Simulation");
        setSize(500, 200);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLocationRelativeTo(null);

        // Start the thread
        t = new Thread(this);
        t.start();

        setVisible(true);
    }

    @Override
    public void run() {
        try {
            while (true) {
                // Increment X to move text from left to right
                x += 5;

                // Reset position if it goes beyond the frame width
                if (x > getWidth()) {
                    x = -200; // Start slightly off-screen to the left for smooth entry
                }

                repaint(); // Refresh the screen
                Thread.sleep(50); // Control the speed of scrolling
            }
        } catch (InterruptedException e) {
            System.out.println(e);
        }
    }

    @Override
    public void paint(Graphics g) {
        super.paint(g);
        g.setFont(new Font("Arial", Font.BOLD, 20));
        g.setColor(Color.BLUE);
        
        // Draw the string at the updated x position
        g.drawString(message, x, y);
    }

    public static void main(String[] args) {
        new ScrollingText();
    }
}

        setTitle("College Information System");
        setSize(600, 400);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLocationRelativeTo(null);

        // Define Column Names
        String[] columns = {"College ID", "College Name", "Address", "Established Year"};
        
        // Initialize Table Model (0 rows to start)
        model = new DefaultTableModel(columns, 0);
        table = new JTable(model);

        // Add Table to ScrollPane (essential for headers and scrolling)
        JScrollPane scrollPane = new JScrollPane(table);
        add(scrollPane, BorderLayout.CENTER);

        // Fetch and Display Data
        fetchCollegeData();

        setVisible(true);
    }

    public void fetchCollegeData() {
        Connection conn = null;
        Statement st = null;
        ResultSet rs = null;

        try {
            // Load PostgreSQL Driver
            Class.forName("org.postgresql.Driver");
            
            // Connect to Database
            conn = DriverManager.getConnection("jdbc:postgresql://localhost/postgres", "postgres", "password");
            
            st = conn.createStatement();
            rs = st.executeQuery("SELECT * FROM College");

            // Process ResultSet and add rows to the model
            while (rs.next()) {
                int id = rs.getInt("CID");
                String name = rs.getString("CName");
                String addr = rs.getString("address");
                int year = rs.getInt("Year");

                // Add row to the model
                model.addRow(new Object[]{id, name, addr, year});
            }

        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, "Error: " + e.getMessage());
        } finally {
            try {
                if (rs != null) rs.close();
                if (st != null) st.close();
                if (conn != null) conn.close();
            } catch (SQLException se) {
                se.printStackTrace();
            }
        }
    }

    public static void main(String[] args) {
        // Use invokeLater for thread-safe UI creation
        SwingUtilities.invokeLater(() -> new CollegeTableDisplay());
    }

        g2.setStroke(new BasicStroke(2));

        // Draw Flag Pole
        g2.setColor(Color.BLACK);
        g2.fillRect(50, 50, 10, 400);

        // Draw Saffron Band
        if (showSaffron) {
            g2.setColor(new Color(255, 153, 51)); // Saffron
            g2.fillRect(60, 50, 300, 60);
        }

        // Draw White Band
        if (showWhite) {
            g2.setColor(Color.WHITE);
            g2.fillRect(60, 110, 300, 60);
            g2.setColor(Color.BLACK);
            g2.drawRect(60, 110, 300, 60); // Border for white
        }

        // Draw Green Band
        if (showGreen) {
            g2.setColor(new Color(19, 136, 8)); // Green
            g2.fillRect(60, 170, 300, 60);
        }

        // Draw Ashoka Chakra
        if (showChakra) {
            g2.setColor(new Color(0, 0, 128)); // Navy Blue
            int cx = 210 - 25; // Center X
            int cy = 140 - 25; // Center Y
            g2.drawOval(185, 115, 50, 50); // Circle
            
            // Draw 24 spokes
            for (int i = 0; i < 24; i++) {
                double angle = Math.toRadians(i * 15);
                int x2 = (int) (210 + 25 * Math.cos(angle));
                int y2 = (int) (140 + 25 * Math.sin(angle));
                g2.drawLine(210, 140, x2, y2);
            }
        }
    }
}
