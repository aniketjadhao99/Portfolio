                    try {
                            BufferedReader br = new BufferedReader(new FileReader(file));
                            String line;
                            int lineNo = 0;
                            while ((line = br.readLine()) != null) {
                                    lineNo++;
                                    if (line.contains(searchWord)) {
                                            System.out.println("Found in File: " + file.getName() +
                                            " at Line: " + lineNo);
                                        }
                                    }
                                    br.close();
                                } catch (Exception e) {
                                    System.out.println(e);
                                }
                            }
                        }
